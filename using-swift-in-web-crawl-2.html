import SwiftUI

struct ContentView: View {
    var body: some View {
        Text("Hello, world!")
            .onTapGesture { self.startLoad() }
    }
    
    func startLoad() {
        // use guard let statement to ensure url isn't an optional
        guard let url = URL(string: "https://rate.bot.com.tw/xrt?Lang=zh-TW") else {
            print("Invalid URL!")
            return
        }
        
        // use our url to customize an URL request
        let request = URLRequest(url: url)
        
        // use the customized request to create an URL session's data task, and define the completion closure
        let task = URLSession.shared.dataTask(with: request) { (data, response, error) in
            if let response = response {
                if let mimeType = response.mimeType {
                    print("mimeType: \(mimeType)")
                }
            }
            
            // if request if failed, print the error message
            if let error = error {
                print("error: \(error.localizedDescription)")
                return
            }
            
            // if data does exist, convert it to string and print out
            if let data = data {
                if let html = String(data: data, encoding: .utf8) {
                    // we have good data, go back to main thread
                    DispatchQueue.main.async {
                        print("html: \(html)")
                    }
                }
                return
            }
        }
        
        // start our URL session's data task
        task.resume()
    }
}
